package day7;

import java.util.Scanner;

public class test2 {
	public static void main(String[] args) {
		//2. 숫자를 두개 입력받고, 두 수의 합이 10이 넘을 경우 
		//'큰 수' 아닐 경우 '작은 수' 출력

		Scanner s = new Scanner(System.in);
		System.out.print("숫자를 입력해주세요");
		int num = s.nextInt();
		System.out.print("숫자를 입력해주세요");
		int num2 = s.nextInt();
		
		if(num+num2 <10) {
			System.out.println("작은수");
		}
		else {
			System.out.println("큰수");
		}
	}

}
